\section{Certificate Synthesis}
\label{sec:synthesis}
In practice, we must synthesize certificates automatically. This section presents an SMT-based method using counterexample-guided inductive synthesis. We apply counterexample-guided inductive synthesis (CEGIS)~\cite{solar2008program} to automatically construct both transition safety certificates and transition persistence certificates. The core idea is to iteratively refine candidate certificates by sampling states and using SMT solvers to satisfy certificate conditions on these samples.

\textbf{Transition Safety Certificates.} Given an accepting state $q_k \in Acc^{start}$ in NBA $\mathcal{A} = (\Sigma, Q, q_0, \delta, Acc)$, we define a parametric template:
\begin{align}
B_k(x, q; \mathbf{c}) = \sum_{i=1}^{m} c_i \, p_i(x, q),
\end{align}
where $\{p_1, \ldots, p_m\}$ is a user-defined basis of monomials over $\mathcal{X} \times Q$ with degree at most $d$, and $\mathbf{c} = (c_1, \ldots, c_m) \in \mathbb{R}^m$ are unknown coefficients. Our goal is to find $\mathbf{c}$ such that $B_k(\cdot, \cdot; \mathbf{c})$ satisfies conditions~(\ref{btsc1}), (\ref{btsc2}), and~(\ref{btsc3}). Sample finite sets $S_0 \subseteq \mathcal{X}_0 \times \{q_0\}$, $S_{\mathcal{X}} \subseteq \mathcal{X} \times Q$, and $S_u \subseteq \mathcal{X} \times \{q_k\}$. The SMT formula for candidate generation is:
\begin{align}
  \Phi_{\text{sample}} \triangleq \;
  &\bigwedge_{(x,q) \in S_0} B_k(x, q; \mathbf{c}) \geq 0 \label{smt:init} \\
  \wedge \; &\bigwedge_{\substack{(x,q) \in S_{\mathcal{X}} \\ (q, \mathfrak{L}(x), q') \in \delta}} \big[ B_k(x, q; \mathbf{c}) \geq 0 \Rightarrow B_k(f(x), q'; \mathbf{c}) \geq 0 \big] \label{smt:inv} \\
  \wedge \; &\bigwedge_{(x,q) \in S_u} B_k(x, q; \mathbf{c}) < 0. \label{smt:unsafe}
\end{align}
If satisfiable with solution $\hat{\mathbf{c}}$, define $\hat{B}_k = B_k(\cdot, \cdot; \hat{\mathbf{c}})$ and verify by checking:
\begin{align}
  \Phi_{\text{init}}^{\neg} &\triangleq x \in \mathcal{X}_0 \land \hat{B}_k(x, q_0) < 0, \label{smt:cex-init} \\
  \Phi_{\text{inv}}^{\neg} &\triangleq x \in \mathcal{X} \land (q, \mathfrak{L}(x), q') \in \delta \land \hat{B}_k(x, q) \geq 0 \land \hat{B}_k(f(x), q') < 0, \label{smt:cex-inv} \\
  \Phi_{\text{unsafe}}^{\neg} &\triangleq x \in \mathcal{X} \land \hat{B}_k(x, q_k) \geq 0. \label{smt:cex-unsafe}
\end{align}
Counterexamples are added to $S_0$, $S_{\mathcal{X}}$, or $S_u$ accordingly and the process iterates until no counterexamples are found, yielding a valid transition safety certificate.

\textbf{Transition Persistence Certificates.} For each accepting transition pair $(q_k, q_l) \in Acc^{pair}$, we define a template:
\begin{align}
B_{k,l}(x; \mathbf{c}) = \sum_{i=1}^{m'} c_i \, p_i(x),
\end{align}
where $\{p_1, \ldots, p_{m'}\}$ are monomials over $\mathcal{X}$ and $\mathbf{c} \in \mathbb{R}^{m'}$. We seek $\mathbf{c}$ and $\epsilon > 0$ satisfying conditions~(\ref{btpc1}), (\ref{btpc2}), and~(\ref{btpc3}). Sample finite sets $S_0 \subseteq \mathcal{X}_0$ and $S_{\mathcal{X}} \subseteq \mathcal{X}$. The SMT formula for candidate generation is:
\begin{align}
  \Psi_{\text{sample}} \triangleq \;
  &\bigwedge_{x \in S_0} B_{k,l}(x; \mathbf{c}) \geq 0 \label{smt:tp-init} \\
  \wedge \; &\bigwedge_{x \in S_{\mathcal{X}}} \big[ B_{k,l}(x; \mathbf{c}) \geq 0 \Rightarrow B_{k,l}(f(x); \mathbf{c}) \geq 0 \big] \label{smt:tp-inv} \\
  \wedge \; &\bigwedge_{\substack{x \in S_{\mathcal{X}} \\ (q_k, \mathfrak{L}(x), q_l) \notin Acc^{k,l}}} B_{k,l}(x; \mathbf{c}) \geq B_{k,l}(f(x); \mathbf{c}) \label{smt:tp-non-dec} \\
  \wedge \; &\bigwedge_{\substack{x \in S_{\mathcal{X}} \\ (q_k, \mathfrak{L}(x), q_l) \in Acc^{k,l}}} B_{k,l}(x; \mathbf{c}) \geq B_{k,l}(f(x); \mathbf{c}) + \epsilon. \label{smt:tp-dec}
\end{align}
If satisfiable with solution $(\hat{\mathbf{c}}, \hat{\epsilon})$, define $\hat{B}_{k,l} = B_{k,l}(\cdot; \hat{\mathbf{c}})$ and verify by checking:
\begin{align}
  \Psi_{\text{init}}^{\neg} &\triangleq x \in \mathcal{X}_0 \land \hat{B}_{k,l}(x) < 0, \label{smt:tp-cex-init} \\
  \Psi_{\text{inv}}^{\neg} &\triangleq x \in \mathcal{X} \land \hat{B}_{k,l}(x) \geq 0 \land \hat{B}_{k,l}(f(x)) < 0, \label{smt:tp-cex-inv} \\
  \Psi_{\text{dec}}^{\neg} &\triangleq x \in \mathcal{X} \land (q_k, \mathfrak{L}(x), q_l) \in Acc^{k,l} \land \hat{B}_{k,l}(x) < \hat{B}_{k,l}(f(x)) + \hat{\epsilon}, \label{smt:tp-cex-dec} \\
  \Psi_{\text{non-dec}}^{\neg} &\triangleq x \in \mathcal{X} \land (q_k, \mathfrak{L}(x), q_l) \notin Acc^{k,l} \land \hat{B}_{k,l}(x) < \hat{B}_{k,l}(f(x)). \label{smt:tp-cex-non-dec}
\end{align}
Counterexamples are added to samples and the process iterates until no counterexamples are found, yielding a valid transition persistence certificate.
