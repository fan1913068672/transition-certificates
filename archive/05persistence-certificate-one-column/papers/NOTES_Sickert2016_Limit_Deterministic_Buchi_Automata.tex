\section{Core Contribution}

This paper presents a \textbf{direct, compositional construction} from LTL formulas to limit-deterministic Büchi automata (LDBA):

\begin{enumerate}
    \item \textbf{Two-component structure}:
    \begin{itemize}
        \item Non-deterministic initial component: guesses set of eventually true G-subformulas
        \item Deterministic accepting component: verifies the guess
    \end{itemize}

    \item \textbf{Key advantage over DRA}: LDBA often much smaller while supporting same verification tasks

    \item \textbf{Quantitative MDP verification}: Due to special structure, can use standard algorithm (same as DRA) without modification

    \item \textbf{Complexity}: Double exponential (worst case), matching DRA

    \item \textbf{Performance}: Much smaller automata for formulas with deep nesting of modal operators
\end{enumerate}

\section{Preliminaries}

\subsection{LTL Syntax}

\begin{definition}[LTL in Negation Normal Form]
\begin{equation}
\varphi ::= \mathtt{tt} \mid \mathtt{ff} \mid a \mid \neg a \mid \varphi \land \varphi \mid \varphi \lor \varphi \mid \mathbf{X}\varphi \mid \mathbf{F}\varphi \mid \mathbf{G}\varphi \mid \varphi \mathbf{U}\varphi
\end{equation}
\end{definition}

\textbf{Note}: No Release operator $\mathbf{R}$, but both $\mathbf{F}$ and $\mathbf{G}$ included. Negations pushed inside using $\neg(\varphi \mathbf{U}\psi) = (\neg\psi \mathbf{U}(\neg\psi \land \neg\varphi)) \lor \mathbf{G}\neg\psi$.

\subsection{After Function}

\begin{definition}[After Function]
For formula $\varphi$ and letter $\nu \in 2^{Ap}$, $af(\varphi, \nu)$ is defined as:
\begin{align}
af(\mathtt{tt}, \nu) &= \mathtt{tt} & af(\mathtt{ff}, \nu) &= \mathtt{ff}\\
af(a, \nu) &= \begin{cases}\mathtt{tt} & a \in \nu\\ \mathtt{ff} & a \notin \nu\end{cases} & af(\neg a, \nu) &= \begin{cases}\mathtt{ff} & a \in \nu\\ \mathtt{tt} & a \notin \nu\end{cases}\\
af(\varphi \land \psi, \nu) &= af(\varphi, \nu) \land af(\psi, \nu) & af(\varphi \lor \psi, \nu) &= af(\varphi, \nu) \lor af(\psi, \nu)\\
af(\mathbf{X}\varphi, \nu) &= \varphi\\
af(\mathbf{G}\varphi, \nu) &= af(\varphi, \nu) \land \mathbf{G}\varphi\\
af(\mathbf{F}\varphi, \nu) &= af(\varphi, \nu) \lor \mathbf{F}\varphi\\
af(\varphi \mathbf{U}\psi, \nu) &= af(\psi, \nu) \lor (af(\varphi, \nu) \land \varphi \mathbf{U}\psi)
\end{align}
\end{definition}

\textbf{Key property}: $ww' \models \varphi$ iff $w' \models af(\varphi, w)$ (Lemma 1).

\begin{definition}[Reachable Formulas]
\begin{equation}
\text{Reach}(\varphi) = \{[\psi]_P \mid \exists w. \psi = af(\varphi, w)\}
\end{equation}
where $[\psi]_P$ is the equivalence class under propositional equivalence.
\end{definition}

\begin{lemma}
For formula $\varphi$ with $n$ proper subformulas, $|\text{Reach}(\varphi)| \leq 2^{2^n}$.
\end{lemma}

\subsection{Limit-Deterministic Büchi Automata}

\begin{definition}[Transition-based Generalized Büchi Automaton (TGBA)]
A TGBA is $\mathcal{B} = (\Sigma, Q, \Delta, q_0, \alpha)$ where:
\begin{itemize}
    \item $\Sigma$: alphabet
    \item $Q$: finite set of states
    \item $\Delta: Q \times \Sigma \to 2^Q$: transition function
    \item $q_0$: initial state
    \item $\alpha = \{F_1, \ldots, F_n\}$ with $F_i \subseteq Q \times \Sigma \times Q$: accepting condition
\end{itemize}

A run is accepting if $\inf(r) \cap F \neq \emptyset$ for each $F \in \alpha$.
\end{definition}

\begin{definition}[Limit-Determinism]
A TGBA $\mathcal{B} = (\Sigma, Q, \Delta, q_0, \alpha)$ is \emph{limit-deterministic} if $Q = Q_N \sqcup Q_D$ such that:
\begin{enumerate}
    \item $\Delta(q, \nu) \subseteq Q_D$ and $|\Delta(q, \nu)| = 1$ for every $q \in Q_D$, $\nu \in \Sigma$
    \item $F \subseteq Q_D \times \Sigma \times Q_D$ for all $F \in \alpha$
\end{enumerate}
\end{definition}

\textbf{Intuition}: Can split into non-deterministic component $Q_N$ (no accepting transitions) and deterministic component $Q_D$. After jumping to $Q_D$, must stay there forever.

\section{Construction Overview}

For formula $\varphi = c \lor \mathbf{X}\mathbf{G}(a \lor \mathbf{F}b)$, the automaton has two components:

\subsection{Initial Component (Non-deterministic)}

\begin{itemize}
    \item States: $\text{Reach}(\varphi)$
    \item Initial state: $\varphi$
    \item Transitions: For state $\varphi'$ and letter $\nu$, transition to $af(\varphi', \nu)$
    \item $\epsilon$-jumps: From state $\varphi'$, can jump to accepting component for any subset $\mathcal{G}$ of G-subformulas
\end{itemize}

\textbf{Key idea}: The $\epsilon$-jump guesses which G-subformulas are eventually true.

\subsection{Accepting Component (Deterministic)}

For each subset $\mathcal{G} \subseteq \mathbb{G}$ of G-subformulas, construct product of:

\begin{enumerate}
    \item \textbf{G-monitors}: One for each $\mathbf{G}\psi \in \mathcal{G}$, verifying $\mathbf{G}(\psi[\mathcal{G}])$ holds
    \item \textbf{Auxiliary monitor}: Verifies $\varphi'[\mathcal{G}]$ holds (where $\varphi'$ is source state of $\epsilon$-jump)
\end{enumerate}

\begin{definition}[Formula Substitution]
Given formula $\varphi$ and set $\mathcal{G}$ of G-formulas, $\varphi[\mathcal{G}]$ substitutes:
\begin{itemize}
    \item $\mathtt{tt}$ for every G-subformula in $\mathcal{G}$
    \item $\mathtt{ff}$ for every other G-subformula
\end{itemize}
\end{definition}

\textbf{Correctness intuition}: If $\mathbf{G}\psi \in \mathcal{G}$ holds from current point, it always holds (replace by $\mathtt{tt}$). If $\mathbf{G}\psi \notin \mathcal{G}$, check $\varphi$ holds even if $\mathbf{G}\psi$ never holds (replace by $\mathtt{ff}$).

\section{G-Monitor Construction}

For G-free formula $\psi$, construct deterministic Büchi automaton $\mathcal{U}(\mathbf{G}\psi)$ recognizing $L(\mathbf{G}\psi)$:

\begin{definition}[G-Monitor]
\begin{equation}
\mathcal{U}(\mathbf{G}\psi) = (2^{Ap}, \text{Reach}(\psi) \times \text{Reach}(\psi), \delta, (\psi, \mathtt{tt}), F)
\end{equation}
where:
\begin{align}
\delta((\xi, \zeta), \nu) &= \begin{cases}
(af(\zeta, \nu) \land \psi, \mathtt{tt}) & \text{if } af(\xi, \nu) \equiv_P \mathtt{tt}\\
(af(\xi, \nu), af(\zeta, \nu) \land \psi) & \text{otherwise}
\end{cases}\\
F &= \{((\xi, \zeta), \nu, p) \mid af(\xi, \nu) \equiv_P \mathtt{tt}\}
\end{align}
\end{definition}

\textbf{Intuition}: States $(\phi_1, \phi_2)$ track two "proof obligations":
\begin{itemize}
    \item $\phi_1$: current obligation
    \item $\phi_2$: obligation on hold
\end{itemize}

When $\phi_1 = \mathtt{tt}$, current obligation satisfied, switch to checking $\phi_2$ (accepting transition). If this happens infinitely often, all suffixes satisfy $\psi$, so word satisfies $\mathbf{G}\psi$.

\begin{lemma}
Let $\psi$ be G-free. Then $w \models \mathbf{G}\psi$ iff $\mathcal{U}(\mathbf{G}\psi)$ accepts $w$.
\end{lemma}

\section{Complete Construction}

\begin{definition}[LDBA for LTL Formula]
Given formula $\varphi$ with G-subformulas $\mathbb{G}$:

\begin{itemize}
    \item \textbf{Initial component}: $\mathcal{N} = (2^{Ap}, \text{Reach}(\varphi), af, \varphi)$

    \item \textbf{Accepting component}: For each $\mathcal{G} \subseteq \mathbb{G}$, construct:
    \begin{equation}
    \mathcal{P}(\mathcal{G}) = (2^{Ap}, \prod_{\mathbf{G}\psi_i \in \mathcal{G}} Q_i, \prod \delta_i, (q_{01}, \ldots, q_{0n}), \{F'_1, \ldots, F'_n\})
    \end{equation}
    where $\mathcal{U}_i$ is G-monitor for $\mathbf{G}(\psi_i[\mathcal{G}])$

    \item \textbf{$\epsilon$-transitions}: From state $\chi \in Q_N$ to $(\chi[\mathcal{G}], q_{0\mathcal{G}})$ for each $\mathcal{G} \subseteq \mathbb{G}$

    \item \textbf{Accepting sets}: $F_\xi = \bigcup_{\mathcal{G} \subseteq \mathbb{G}} \{((\mathtt{tt}, q), \nu, (\mathtt{tt}, q')) \mid (q, \nu, q') \in F_{(\mathcal{G}, \xi)}\}$ for each $\xi \in \mathbb{G}$
\end{itemize}
\end{definition}

\section{Correctness}

\begin{definition}[Stable Set]
A set $\mathcal{G} \subseteq \mathbb{G}$ is \emph{stable} for word $w$ if:
\begin{itemize}
    \item $\forall \mathbf{G}\psi \in \mathcal{G}. \forall i \geq 0. w_i \models \mathbf{G}\psi$
    \item $\forall \mathbf{G}\psi \in \mathbb{G} \setminus \mathcal{G}. \forall i \geq 0. w_i \not\models \mathbf{G}\psi$
\end{itemize}
\end{definition}

\textbf{Key observation}: At most one set is stable for a given word.

\begin{lemma}
Let $\varphi$ be a formula and $w$ a word with stable set $\mathcal{G}$. Then $w \models \varphi$ iff $w \models \varphi[\mathcal{G}]$.
\end{lemma}

\begin{theorem}[Correctness]
Let $w$ be a word. Then $w \models \varphi$ iff $\mathcal{A}$ accepts $w$.
\end{theorem}

\section{Complexity}

\subsection{Upper Bound}

\begin{itemize}
    \item Initial component: $|\text{Reach}(\varphi)| = O(2^{2^n})$
    \item For each $\mathcal{G} \subseteq \mathbb{G}$: accepting component has $|\mathcal{G}|$ G-monitors + 1 auxiliary monitor, each of size $O(2^{2^n})$
    \item Total: $O(2^{2^n}) + 2^{|\mathbb{G}|} \cdot O(2^{2^{n+\log\log n}}) = O(2^{2^{n+\log n + \log\log n}})$
\end{itemize}

\textbf{Conclusion}: Double exponential in formula size.

\subsection{Lower Bound}

\begin{theorem}
There exists a family of formulas $\phi_n$ of size $O(n^2)$ such that the smallest LDBA recognizing $L(\phi_n)$ has at least $2^{2^n}$ states.
\end{theorem}

\textbf{Proof idea}: Use language $L(r_n^\omega)$ where $r_n = \sum_{w \in \{0,1\}^n} \% (0|1|\#)^* \# w \# (0|1|\#)^* \$ w$. This forces automaton to remember exponentially many strings of length $n$.

\section{Quantitative Probabilistic Model Checking}

\textbf{Key result}: Despite non-determinism, these LDBAs can be used for quantitative MDP verification with the standard algorithm.

\subsection{Product Construction}

Given MDP $\mathcal{M} = (S, \text{Act}, P, s_0, Ap, L)$ and LDBA $\mathcal{A}$:

\begin{itemize}
    \item Add $\epsilon$-actions to product: For each $\epsilon$-transition in $\mathcal{A}$ to state $q$, add action $\epsilon_q$ in product
    \item $\epsilon$-action probabilities: $P'((s,q), \epsilon_{\hat{q}}, (s',q')) = \begin{cases}1 & s = s' \land q \xrightarrow{\epsilon} q' = \hat{q}\\ 0 & \text{otherwise}\end{cases}$
\end{itemize}

\begin{theorem}[Quantitative Model Checking]
For any formula $\varphi$, the automaton $\mathcal{A}$ can be used in the standard probabilistic model checking algorithm:
\begin{equation}
\sup_\sigma \mathbb{P}_{\mathcal{M}^\sigma}[L(\mathcal{A})] = \sup_\sigma \mathbb{P}_{(\mathcal{M} \times \mathcal{A})^\sigma}[\Diamond X]
\end{equation}
where $\Diamond X$ is the set of runs reaching an accepting MEC.
\end{theorem}

\textbf{Key proof ingredients}:
\begin{enumerate}
    \item Define $\text{index}(\rho) := \min\{i \mid \forall \mathbf{G}\psi \in \mathbb{G}. w \models \mathbf{F}\mathbf{G}\psi \Rightarrow w_i \models \mathbf{G}\psi\}$
    \item Show that in BSCC of Markov chain, almost all runs have same stable set $\mathcal{G}$ (Lemma 12)
    \item Construct scheduler that waits for BSCC, then chooses correct $\epsilon$-transition based on stable $\mathcal{G}$
    \item Verify two safety conditions:
    \begin{itemize}
        \item Lingering in initial component is safe: $L(\text{init}) \supseteq \bigcup_{a \in \text{All}} L(a)$ (Lemma 9)
        \item Switching at BSCC is safe: $\mathbb{P}_m[L(\text{init})] = \mathbb{P}_m[L(\text{acc})]$
    \end{itemize}
\end{enumerate}

\section{Experimental Results}

Benchmarks compared with:
\begin{itemize}
    \item \textbf{L3 (ltl3ba)}: LTL to NBA, then semi-determinization
    \item \textbf{S (Spot)}: Advanced LTL to NBA with optimizations
    \item \textbf{L2D (ltl2dstar)}: LTL to DRA
    \item \textbf{R (Rabinizer)}: Direct LTL to DRA
    \item \textbf{LD (ltl2ldba)}: This construction
\end{itemize}

\textbf{Key findings}:
\begin{enumerate}
    \item \textbf{Deeply nested formulas}: LD significantly outperforms all others
    \begin{itemize}
        \item $f(2,4) = (\mathbf{GF}a_3)\mathbf{U}(\mathbf{G}f(1,4))$: LD has 92 states, L3 has 12314, S has 14161
    \end{itemize}

    \item \textbf{General formulas}: LD competitive with existing tools
    \begin{itemize}
        \item Total over 359 formulas: LD has 9518 states, S has 29559, L3 has 55444
    \end{itemize}

    \item \textbf{Acceptance sets}: LD requires more acceptance sets (TGBA) but fewer states
\end{enumerate}

\section{Relation to Our Work}

\subsection{Similarities}

\begin{enumerate}
    \item \textbf{Both use transition-based acceptance}: Sickert uses transition-based TGBA, we use transition-based NBA

    \item \textbf{Both have two-component structure}:
    \begin{itemize}
        \item Sickert: Initial (non-det) + Accepting (det)
        \item Our work: Safety certificates + Persistence certificates
    \end{itemize}

    \item \textbf{Both reduce search space}: Compared to full product constructions

    \item \textbf{Both support quantitative verification}: Can be used with standard algorithms
\end{enumerate}

\subsection{Key Differences}

\begin{enumerate}
    \item \textbf{System Model}:
    \begin{itemize}
        \item \textbf{Sickert}: Symbolic state representation via formulas, automata-based
        \item \textbf{Our work}: Continuous state space $\mathcal{X} \subseteq \mathbb{R}^n$, dynamical systems
    \end{itemize}

    \item \textbf{Certificate Definition}:
    \begin{itemize}
        \item \textbf{Sickert}: G-monitors defined on $\text{Reach}(\psi) \times \text{Reach}(\psi)$ (formula pairs)
        \item \textbf{Our work}: Persistence certificates defined on $\mathcal{X}$ alone (continuous state)
    \end{itemize}

    \item \textbf{Guessing vs. Proving}:
    \begin{itemize}
        \item \textbf{Sickert}: Non-deterministically guess which G-formulas eventually hold
        \item \textbf{Our work}: Prove via certificates (safety blocks, persistence ranks)
    \end{itemize}

    \item \textbf{Dimension of Certificates}:
    \begin{itemize}
        \item \textbf{Sickert G-monitor}: Pairs of formulas $(\phi_1, \phi_2)$ from $\text{Reach}(\psi)$
        \item \textbf{Our persistence}: Single function $B_{k,l}(x)$ on continuous space
    \end{itemize}

    \item \textbf{Verification Target}:
    \begin{itemize}
        \item \textbf{Sickert}: Probabilistic systems (MDPs, Markov chains)
        \item \textbf{Our work}: Deterministic continuous systems $\mathfrak{S} = (\mathcal{X}, \mathcal{X}_0, f)$
    \end{itemize}
\end{enumerate}

\subsection{Conceptual Connections}

\begin{enumerate}
    \item \textbf{Breakpoint Construction}: Sickert's G-monitor uses breakpoint construction with states $(\rho_1, \rho_2)$ tracking current and on-hold obligations. This is analogous to our ranking function that decreases with each accepting transition occurrence.

    \item \textbf{Stable Sets}: Sickert's notion of stable set $\mathcal{G}$ (G-formulas that eventually always hold) is conceptually similar to our persistence certificates proving finite occurrence of accepting transitions.

    \item \textbf{Separation of Concerns}:
    \begin{itemize}
        \item Sickert separates: (1) guessing eventual truth, (2) verifying guess
        \item Our work separates: (1) proving unreachability (safety), (2) proving finite occurrence (persistence)
    \end{itemize}
\end{enumerate}

\section{Key Insights for Our Paper}

\subsection{What We Can Learn}

\begin{enumerate}
    \item \textbf{Transition-based acceptance is powerful}: Sickert shows TGBA with transition-based acceptance enables smaller automata

    \item \textbf{Two-component architectures work well}:
    \begin{itemize}
        \item Initial component for exploration/setup
        \item Accepting component for verification
    \end{itemize}

    \item \textbf{G-formulas are special}: Treating G-subformulas separately (Sickert guesses their eventual truth, we rank on state space) is effective

    \item \textbf{Dimension reduction matters}: Both approaches reduce search space by avoiding full product
\end{enumerate}

\section{Summary}

Sickert et al. present a direct, compositional construction from LTL to limit-deterministic Büchi automata. Their approach uses a non-deterministic initial component to guess which G-subformulas eventually hold, then verifies this guess using deterministic G-monitors defined on formula pairs. While operating in a different domain (symbolic/probabilistic vs. continuous/deterministic), their work demonstrates the power of transition-based acceptance, two-component architectures, and treating G-formulas specially—concepts that resonate with our persistence certificate approach on continuous state spaces.
